    Text¶

    JS and TS support multiline strings, but can't strip indentation.

    char character = 'a';

Character boxedCharacter = 'a';

String string = "Hello";

String multilineString =
    """
    Hello
    World
    """;

    char character = 'a'

var string = 'Hello'

var multilineString =
    '''
    Hello
    World
    '''.stripIndent().trim()

    val character = 'a'

val string = "Hello"

val multilineString =
    """
    Hello
    World
    """.trimIndent()

    from textwrap import *

string = 'Hello'

multiline_string = \
    dedent(
        '''
        Hello
        World
        ''',
    ).strip()

    let string = 'Hello';

    let string: string = 'Hello';
